a )Reescreva a função abaixo usando retornos antecipados.

public boolean podeAcessarSistema(Usuario usuario) {
    if (usuario != null) {
        if (usuario.isAtivo()) {
            if (!usuario.isBloqueado()) {
                if (usuario.temPermissao("SISTEMA")) {
                    return true;
                }
            }
        }
    }

    return false;
}

    }
